```js
interface PagesOptions {
  /**
   * Page format that determines page size and margins.
   * Available formats: A4, A3, A5, Letter, Legal, Tabloid
   * Each format has predefined dimensions and margins.
   * @default 'A4'
   */
  pageFormat: 'A4' | 'A3' | 'A5' | 'Letter' | 'Legal' | 'Tabloid' | string

  /**
   * Height of the page header in pixels.
   * @default 50
   */
  headerHeight?: number

  /**
   * Height of the page footer in pixels.
   * @default 50
   */
  footerHeight?: number

  /**
   * Gap between pages in pixels.
   * @default 50
   */
  pageGap?: number

  /**
   * Footer content for the right side. Can be a string or a function.
   * @default ''
   */
  footerRight: string | ((pageNumber: number, totalPages: number) => string)

  /**
   * Footer content for the left side. Can be a string or a function.
   * @default ''
   */
  footerLeft: string | ((pageNumber: number, totalPages: number) => string)

  /**
   * Footer content for the center. Can be a string or a function.
   * @default ''
   */
  footerCenter: string | ((pageNumber: number, totalPages: number) => string)

  /**
   * Header content for the right side. Can be a string or a function.
   * @default ''
   */
  headerRight: string | ((pageNumber: number, totalPages: number) => string)

  /**
   * Header content for the left side. Can be a string or a function.
   * @default ''
   */
  headerLeft: string | ((pageNumber: number, totalPages: number) => string)

  /**
   * Header content for the center. Can be a string or a function.
   * @default ''
   */
  headerCenter: string | ((pageNumber: number, totalPages: number) => string)

  /**
   * Background color for the page break area.
   * @default '#fff'
   */
  pageBreakBackground?: string
}
```

| Option                | Type / Values                | Default   | Description                                                                                 |
|-----------------------|------------------------------|-----------|---------------------------------------------------------------------------------------------|
| `pageFormat`          | `'A4' \| 'A3' \| 'A5' \| 'Letter' \| 'Legal' \| 'Tabloid' \| string` | `'A4'`    | Predefined page format. Margins and size are fixed.                                          |
| `headerHeight`        | `number` (px)                | `50`      | Header height in pixels                                                                     |
| `footerHeight`        | `number` (px)                | `50`      | Footer height in pixels                                                                     |
| `pageGap`             | `number` (px)                | `50`      | Gap between pages in pixels                                                                 |
| `headerLeft`          | `string \| fn`               | `''`      | Header left content; string (supports `{page}`/`{total}`) or (page, total) => string      |
| `headerCenter`        | `string \| fn`               | `''`      | Header center content; string (supports `{page}`/`{total}`) or (page, total) => string    |
| `headerRight`         | `string \| fn`               | `''`      | Header right content; string (supports `{page}`/`{total}`) or (page, total) => string     |
| `footerLeft`          | `string \| fn`               | `''`      | Footer left content; string (supports `{page}`/`{total}`) or (page, total) => string      |
| `footerCenter`        | `string \| fn`               | `''`      | Footer center content; string (supports `{page}`/`{total}`) or (page, total) => string    |
| `footerRight`         | `string \| fn`               | `''`| Footer right content; string (supports `{page}`/`{total}`) or (page, total) => string     |
| `pageBreakBackground` | `string` (CSS color)         | `'#fff'`  | Background color for page breaks                                                            |

## Example usage

```js
import { Pages } from '@tiptap-pro/extension-pages@alpha'

const editor = new Editor({
  extensions: [
    Pages.configure({
      pageFormat: 'A4',
      headerHeight: 60,
      footerHeight: 40,
      pageGap: 40,
      footerRight: (page, total) => `Page {page} of {total}`,
      headerCenter: 'My Document',
      pageBreakBackground: '#f8f8f8',
    }),
  ],
})
```
